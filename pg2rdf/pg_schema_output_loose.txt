CREATE GRAPH TYPE NewGraphSchema LOOSE {
  (ForumType: Forum {id DOUBLE, title STRING, creationDate DATE}),
  (Tag_TagClassType: Tag | TagClass {name STRING, id DOUBLE, url STRING}),
  (Organisation_PlaceType: Organisation | Place {type STRING, name STRING, id DOUBLE, url STRING}),
  (PostType: Post {browserUsed STRING, id DOUBLE, locationIP STRING, length INTEGER, creationDate DATE, imageFile STRING, content STRING, language STRING}),
  (PersonType: Person {browserUsed STRING, lastName STRING, firstName STRING, gender STRING, id DOUBLE, locationIP STRING, creationDate DATE, birthday DATE}),
  (CommentType: Comment {browserUsed STRING, content STRING, id DOUBLE, locationIP STRING, length INTEGER, creationDate DATE}),
  (:ForumType)-[CONTAINER_OFType: CONTAINER_OF]->(:PostType),
  (:PersonType|PostType)-[HAS_CREATORType: HAS_CREATOR]->(:PersonType|CommentType),
  (:PersonType)-[HAS_INTERESTType: HAS_INTEREST]->(:TagType),
  (:ForumType)-[HAS_MEMBERType: HAS_MEMBER {joinDate DATE}]->(:PersonType),
  (:ForumType)-[HAS_MODERATORType: HAS_MODERATOR]->(:PersonType),
  (:ForumType|CommentType|PostType)-[HAS_TAGType: HAS_TAG]->(:TagType),
  (:TagType)-[HAS_TYPEType: HAS_TYPE]->(:TagClassType),
  (:CommentType|OrganisationType|PersonType|PostType)-[IS_LOCATED_INType: IS_LOCATED_IN]->(:PlaceType),
  (:PlaceType)-[IS_PART_OFType: IS_PART_OF]->(:PlaceType),
  (:TagClassType)-[IS_SUBCLASS_OFType: IS_SUBCLASS_OF]->(:TagClassType),
  (:PersonType)-[KNOWSType: KNOWS {creationDate DATE}]->(:PersonType),
  (:PersonType)-[LIKESType: LIKES {creationDate DATE}]->(:CommentType|PostType),
  (:CommentType)-[REPLY_OFType: REPLY_OF]->(:CommentType|PostType),
  (:PersonType)-[STUDIES_ATType: STUDIES_AT {classYear INTEGER}]->(:OrganisationType),
  (:PersonType)-[WORKS_ATType: WORKS_AT {workFrom INTEGER}]->(:OrganisationType),
}
